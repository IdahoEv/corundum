def resolve_configuration
  if field_unset?(:words)
    self.words = WORD_SETS.fetch(type.to_s) do
      raise "Word set #{type.inspect} unknown. Choose from: #{WORD_SETS.keys.inspect}"
    end
  end
  super
end
