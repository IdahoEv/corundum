def file_and_line
  @file_and_line ||=
    begin
      if line.nil?
        file
      else
        [file, line].join(":")
      end
    end
end

def to_s(column_widths=nil)
  column_widths ||= {}
  [:file_and_line, :label, :value].map do |name|
    if column_widths.has_key?(name)
      self.send(name).to_s.ljust(column_widths[name])
    else
      self.send(name).to_s
    end
  end.join('  ')
end
