def to_s(widths=nil)
  (passed ? "Ok" : "FAIL") +
  ": #{name} " +
  (summary_counts ? "(#{length})" : "") + "\n" +
    (summary.empty? ? "" : (summary + "\n")) +
    rejects.map do |reject|
    "  " + reject.to_s(widths) + "\n"
    end.join("")
end
