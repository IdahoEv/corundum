def git(*args)
  result = git_command(*args).run
  result.must_succeed!
  result.stdout.lines.to_a
end

def git_command(*args)
  Mattock::CommandLine.new("git", "--no-pager") do |cmd|
    args.each do |arg|
      cmd.options += [*arg]
    end
  end
end
